82  public FieldJson(JSchema schema, string name)
83  : base(name, FieldRole.Value)
84  {
85  _schema = schema;
86  bool optional;
87  var fieldSchema = FieldSchema(name, out optional);
88  var eltSchema = ElementSchema(fieldSchema);
89  ProcessAnnotations(schema, fieldSchema, eltSchema);
90  var fieldName = name.Split('.').Last();
91  JToken date;
92  if (eltSchema.ExtensionData.TryGetValue("DateTime", out date) && date.Value<bool>())
93  {
94  SetType(typeof(DateTime));
95  }
96  else
97  {
98  SetType(eltSchema.Enum.Any() ? null : ToType(eltSchema));
99  }
100  SetAllowsMultiple(fieldSchema.Type.HasValue && fieldSchema.Type.Value.HasFlag(JSchemaType.Array));
101  SetFieldDescription(AString(fieldSchema, "Describe") ?? Language.CamelCase(fieldName));
102  var terms = Strings(fieldSchema, "Terms");
103  JToken maxPhrase;
104  if (terms != null && fieldSchema.ExtensionData.TryGetValue("MaxPhrase", out maxPhrase))
105  {
106  terms = (from seed in terms
107  from gen in Language.GenerateTerms(seed, (int)maxPhrase)
108  select gen).ToArray<string>();
109  }
110  SetFieldTerms(terms ?? Language.GenerateTerms(Language.CamelCase(fieldName), 3));
111  ProcessEnum(eltSchema);
112  SetOptional(optional);
113  SetIsNullable(fieldSchema.Type.HasValue && fieldSchema.Type.Value.HasFlag(JSchemaType.Null));
114  }

143  {

296  protected void ProcessEnum(JSchema schema)
297  {
298  var enums = (from val in schema.Enum select (string)val);
299  var toDescription = new Dictionary<string, string>();
300  var toTerms = new Dictionary<string, string[]>();
301  var toMaxPhrase = new Dictionary<string, int>();
302  JToken values;
303  if (schema.ExtensionData.TryGetValue("Values", out values))
304  {
305  foreach (JProperty prop in values.Children())
306  {
307  var key = prop.Name;
308  if (!enums.Contains(key))
309  {
310  throw new ArgumentException($"{key} is not in enumeration.");
311  }
312  var desc = (JObject)prop.Value;
313  JToken description;
314  if (desc.TryGetValue("Describe", out description))
315  {
316  toDescription.Add(key, (string)description);
317  }
318  JToken terms;
319  if (desc.TryGetValue("Terms", out terms))
320  {
321  toTerms.Add(key, terms.ToObject<string[]>());
322  }
323  JToken maxPhrase;
324  if (desc.TryGetValue("MaxPhrase", out maxPhrase))
325  {
326  toMaxPhrase.Add(key, (int)maxPhrase);
327  }
328  }
329  }
330  foreach (var key in enums)
331  {
332  string description;
333  if (!toDescription.TryGetValue(key, out description))
334  {
335  description = Language.CamelCase(key);
336  }
337  AddDescription(key, description);
338 
339  string[] terms;
340  int maxPhrase;
341  if (!toTerms.TryGetValue(key, out terms))
342  {
343  terms = Language.GenerateTerms(description, 3);
344  }
345  else if (toMaxPhrase.TryGetValue(key, out maxPhrase))
346  {
347  terms = (from seed in terms
348  from gen in Language.GenerateTerms(seed, maxPhrase)
349  select gen).ToArray<string>();
350  }
351  AddTerms(key, terms);
352  }
353  }

393  {

397  var childSchema = property.Value;
398  if (childSchema.Type.HasValue && childSchema.Type.Value.HasFlag(JSchemaType.Array))
399  {
400  if (childSchema.Items.Count() > 0

402  {
403  fields.Add(path);
404  }
405  }
406  else if (IsPrimitiveType(childSchema))
407  {
408  fields.Add(path);
409  }
410  else
411  {
412  Fields(childSchema, path, fields);
413  }
414  }

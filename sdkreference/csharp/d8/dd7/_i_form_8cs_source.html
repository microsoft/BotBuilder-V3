748  private bool MoveToNext(T state, FormState form, NextStep next)
749  {
750  bool found = false;
751  switch (next.Direction)
752  {
753  case StepDirection.Complete:
754  // TODO: Add OnCompletion handler
755  break;
756  case StepDirection.Named:
757  form.StepState = null;
758  if (next.Names.Count() == 0)
759  {
760  goto case StepDirection.Next;
761  }
762  else if (next.Names.Count() == 1)
763  {
764  var name = next.Names.First();
765  var nextStep = -1;
766  for (var i = 0; i < _steps.Count(); ++i)
767  {
768  if (_steps[i].Name() == name)
769  {
770  nextStep = i;
771  break;
772  }
773  }
774  if (nextStep == -1)
775  {
776  throw new ArgumentOutOfRangeException("NextStep", "Does not correspond to a field in the form.");
777  }
778  if (_steps[nextStep].Active(state))
779  {
780  var current = _steps[form.Step];
781  form.SetPhase(_fields.Field(current.Name()).IsUnknown(state) ? StepPhase.Ready : StepPhase.Completed);
782  form.History.Push(form.Step);
783  form.Step = nextStep;
784  form.SetPhase(StepPhase.Ready);
785  found = true;
786  }
787  else
788  {
789  // If we went to a state which is not active fall through to the next active if any
790  goto case StepDirection.Next;
791  }
792  }
793  else
794  {
795  // Always mark multiple names as found so we can handle the user navigation
796  found = true;
797  }
798  break;
799  case StepDirection.Next:
800  var start = form.Step;
801  // Next ready step including current one
802  for (var offset = 0; offset < _steps.Count(); ++offset)
803  {
804  form.Step = (start + offset) % _steps.Count();
805  if (offset > 0)
806  {
807  form.StepState = null;
808  }
809  var step = _steps[form.Step];
810  if ((form.Phase() == StepPhase.Ready || form.Phase() == StepPhase.Responding)
811  && step.Active(state))
812  {
813  if (step.Type() == StepType.Confirm)
814  {
815  // Ensure all dependencies have values
816  foreach (var dependency in step.Dependencies())
817  {
818  var dstep = Step(dependency);
819  var dstepi = StepIndex(dstep);
820  if (dstep.Active(state) && form.Phases[dstepi] != StepPhase.Completed)
821  {
822  form.Step = dstepi;
823  break;
824  }
825  }
826  found = true;
827  }
828  else
829  {
830  found = true;
831  }
832  if (form.Step != start)
833  {
834  form.History.Push(start);
835  }
836  break;
837  }
838  }
839  break;
840  case StepDirection.Previous:
841  while (form.History.Count() > 0)
842  {
843  var lastStepIndex = form.History.Pop();
844  var lastStep = _steps[lastStepIndex];
845  if (lastStep.Active(state))
846  {
847  var step = _steps[form.Step];
848  form.SetPhase(step.Field().IsUnknown(state) ? StepPhase.Ready : StepPhase.Completed);
849  form.Step = lastStepIndex;
850  form.SetPhase(StepPhase.Ready);
851  form.StepState = null;
852  found = true;
853  break;
854  }
855  }
856  break;
857  case StepDirection.Quit:
858  break;
859  case StepDirection.Reset:
860  form.Reset();
861  // TODO: Should I reset the state as well?
862  // Because we redo phase they can go through everything again but with defaults.
863  found = true;
864  break;
865  }
866  return found;
867  }

992  public NextStep Process(Session session, T state, FormState form, string input, IEnumerable<TermMatch> matches, out string prompt)
993  {
994  prompt = null;
995  var iprompt = _field.Prompt();
996  var fieldState = form.StepState as FieldStepState;
997  object response = null;
998  if (fieldState.State == FieldStepStates.SentPrompt)
999  {
1000  // Response to prompt
1001  var firstMatch = matches.FirstOrDefault();
1002  if (matches.Count() == 1)
1003  {
1004  response = SetValue(state, firstMatch.Value);
1005  form.StepState = null;
1006  form.SetPhase(StepPhase.Completed);
1007  }
1008  else if (matches.Count() > 1)
1009  {
1010  // Check multiple matches for ambiguity
1011  var groups = MatchAnalyzer.GroupedMatches(matches);
1012  // 1) Could be multiple match groups like for ingredients.
1013  // 2) Could be overlapping matches like "onion".
1014  // 3) Could be multiple matches where only one is expected.
1015 
1016  // Did you mean {x} or {y}?
1017  var ambiguous = new List<Ambiguous>();
1018  var settled = new List<object>();
1019  foreach (var choices in groups)
1020  {
1021  if (choices.Count() > 1)
1022  {
1023  var unclearMatch = choices.First();
1024  var unclearResponse = input.Substring(unclearMatch.Start, unclearMatch.Length);
1025  var values = from match in choices select match.Value;
1026  ambiguous.Add(new Ambiguous(unclearResponse, values));
1027  }
1028  else
1029  {
1030  var matchValue = choices.First().Value;
1031  if (matchValue.GetType().IsIEnumerable())
1032  {
1033  foreach (var value in matchValue as System.Collections.IEnumerable)
1034  {
1035  settled.Add(value);
1036  }
1037  }
1038  else
1039  {
1040  settled.Add(choices.First().Value);
1041  }
1042  }
1043  }
1044 
1045  if (ambiguous.Count() > 0)
1046  {
1047  // Need 1 or more clarifications
1048  Ambiguous clarify;
1049  form.StepState = new FieldStepState(FieldStepStates.SentClarify) { Settled = settled, Clarifications = ambiguous };
1050  response = SetValue(state, null);
1051  var iChoicePrompt = NextClarifyPrompt(state, form.StepState as FieldStepState, iprompt.Recognizer(), out clarify);
1052  prompt = iChoicePrompt.Prompt(state, _name, clarify.Response);
1053  }
1054  else
1055  {
1056  if (_field.AllowsMultiple())
1057  {
1058  response = SetValue(state, settled);
1059  }
1060  else
1061  {
1062  // Set the resulting values.
1063  // TODO: what about when multiple values are not allowed...
1064  foreach (var value in settled)
1065  {
1066  response = SetValue(state, value);
1067  }
1068  }
1069  form.SetPhase(StepPhase.Completed);
1070  }
1071  }
1072  }
1073  else if (fieldState.State == FieldStepStates.SentClarify)
1074  {
1075  Ambiguous clarify;
1076  var iChoicePrompt = NextClarifyPrompt(state, fieldState, iprompt.Recognizer(), out clarify);
1077  var choices = MatchAnalyzer.Coalesce(iChoicePrompt.Recognizer().Matches(input));
1078  if (choices.Count() == 1)
1079  {
1080  // Clarified ambiguity
1081  fieldState.Settled.Add(choices.First().Value);
1082  fieldState.Clarifications.Remove(clarify);
1083  Ambiguous newClarify;
1084  var newiChoicePrompt = NextClarifyPrompt(state, fieldState, iprompt.Recognizer(), out newClarify);
1085  if (newiChoicePrompt != null)
1086  {
1087  prompt = newiChoicePrompt.Prompt(state, _name, newClarify.Response);
1088  }
1089  else
1090  {
1091  // No clarification left, so set the field
1092  if (_field.AllowsMultiple())
1093  {
1094  response = SetValue(state, fieldState.Settled);
1095  }
1096  else
1097  {
1098  // Set the resulting values.
1099  // TODO: what about when multiple values are not allowed...
1100  foreach (var value in fieldState.Settled)
1101  {
1102  response = SetValue(state, value);
1103  }
1104  }
1105  fieldState = null;
1106  form.SetPhase(StepPhase.Completed);
1107  }
1108  }
1109  }
1110  return _field.Next(response, state);
1111  }
